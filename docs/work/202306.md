  console.log(next, "next");
  console.log(1);
  next();
  console.log(1.1);

middleware.push((next) => {
  console.log(2);
  next();
  console.log(2.1);
});
middleware.push((next) => {
  console.log(3);
  next();
  console.log(3.1);
});
function compose(arr) {
  return (next) => {
    return dispatch(0);
    function dispatch(i) {
      let fn = arr[i];
      if (!fn) return Promise.resolve();
      if (i === arr.length) fn = next;
      try {
        return Promise.resolve(fn(dispatch.bind(null, i + 1)));
      } catch (e) {
        return Promise.reject(e);
      }
    }
  };
}
let fn = compose(middleware);
fn();
// 1 2 3 3.1 2.1 1.1

### TS 练习， PickByType

type OnlyBoolean = PickByType<

type PickByType<T extends Record<PropertyKey, any>, K> = {
  [U in keyof T as T[U] extends K ? U : never]: T[U];

- 练习 Ts PartialByKeys

实现 `PartialByKeys<T, K>`，使 K 匹配的 Key 变成可选的定义，如果不传 K 效果与 Partial 一样：

```ts
interface User {
  name: string;
  age: number;
  address: string;
}

type TT3 = Partial<User>;
type UserPartialName = PartialByKeys<User, "name">; // { name?:string; age:number; address:string }

type PartialByKeys<T, K = keyof T> = {
  [P in keyof T as P extends K ? P : never]?: T[P];
} & {
  [P in keyof T as P extends K ? never : P]: T[P];
} extends infer R
  ? {
      [P in keyof R]: R[P];
    }
  : never;
```
