class LRU {
  constructor(num) {
    this.cache = new Map();
    this.limit = num;
  }
  get(key) {
    if (this.cache.has(key)) {
      const val = this.cache.get(key);
      this.cache.delete(key);
      this.cache.set(key, val);
      return val;
    }
    return -1;
  }
  set(key, val) {
    if (this.cache.has(key)) {
      this.cache.delete(key);
    } else {
      // 需要删除头部的key
      // 删除的前提是池子满了
      if (this.limit <= this.cache.size) {
        var oldKey = this.cache.keys().next().value;
        this.cache.delete(oldKey);
      }
    }
    this.cache.set(key, val);
  }
}

function curry(fn) {
  var judge = (...arg) => {
    if (fn.length === arg.length) {
      return fn(...arg);
    }
    return (...args) => judge(...args, ...arg);
  };
  return judge;
}

function hexToRgb(str) {
  var r, g, b;
  var rgb = str.replace("#", "0x");
  r = rgb >> 16;
  g = (rgb >> 8) & 0xff;
  b = rgb & 0xff;
  return `rgb(${r}, ${g}, ${b})`;
}

function rgbToHex(str) {

    const hex = (+num).toString(16);
    return hex.length === 1 ? `0${hex}` : hex;

}

Array.prototype.myReduce = function (fn, init) {
  const arr = Array.prototype.slice.call(this) || [];
  let res = init ? init : arr[0];
  for (let i = init ? 0 : 1; i < arr.length; i++) {
    res = fn.call(null, res, arr[i], i, arr);

  return res;
